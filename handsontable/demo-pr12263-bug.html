PR #12263 — HotTableRef typing guide for React (bug: old pattern)

Page: Bug (old pattern) CDN v17.0.1 Fix (correct pattern) local build
PR #12263 is documentation-only — it documents the correct HotTableRef typing for React. The runtime behavior is unchanged. The "bug" is incorrect documentation/typing guidance. Since HotTable became a functional component in v14.1.0, useRef<HotTable> no longer gives access to hotInstance. The tabs below show the outdated patterns.
Direct reference (always worked)
// Vanilla JS: direct reference to Handsontable instance
const container = document.getElementById('hot-container');
const hot = new Handsontable(container, { /* config */ });

// Access instance directly
hot.selectCell(0, 0);
const value = hot.getDataAtCell(0, 0);
console.log(value);
OUTDATED PATTERN (before PR #12263 docs)
// ❌ OUTDATED: This pattern is documented in old guides
// HotTable is a functional component since v14.1.0
import { HotTable } from '@handsontable/react-wrapper';
import { useRef } from 'react';

const hotRef = useRef<HotTable>(null);
// ❌ hotRef.current is the functional component instance proxy
// ❌ hotRef.current.hotInstance → undefined (class-based access)
// This worked pre-v14.1.0 when HotTable was a class component
BROKEN pattern (before docs fix)
import { HotTable } from '@handsontable/react-wrapper';
import { useRef } from 'react';

// ❌ WRONG: Using HotTable as the ref type
const hotRef = useRef<HotTable>(null);

function App() {
  const handleClick = () => {
    // ❌ This returns undefined since v14.1.0
    const hotInstance = hotRef.current?.hotInstance;
    console.log(hotInstance); // undefined ❌
    hotInstance?.selectCell(0, 0); // TypeError: Cannot read properties of undefined
  };
  return <HotTable ref={hotRef} data={[['A1']]} licenseKey="non-commercial-and-evaluation" />;
}
Vue: templateRef access
<template>
  <HotTable ref="hotTableRef" :data="data" license-key="non-commercial-and-evaluation" />
  <button @click="selectCell">Select cell (0,0)</button>
</template>
<script setup lang="ts">
import { ref } from 'vue';
import { HotTable } from '@handsontable/vue3';
const hotTableRef = ref(null);
function selectCell() {
  // Vue: access via .hotInstance on the component ref
  hotTableRef.value?.hotInstance?.selectCell(0, 0);
}
</script>
Angular: @ViewChild access
import { Component, ViewChild } from '@angular/core';
import { HotTableComponent } from '@handsontable/angular-wrapper';

@Component({
  selector: 'app-root',
  template: `
    <hot-table #hotTable [data]="data" licenseKey="non-commercial-and-evaluation"></hot-table>
    <button (click)="selectCell()">Select cell (0,0)</button>
  `
})
export class AppComponent {
  @ViewChild('hotTable') hotTable: HotTableComponent;
  data = [['A1', 'B1'], ['A2', 'B2']];
  selectCell() {
    this.hotTable.hotInstance?.selectCell(0, 0);
  }
}
